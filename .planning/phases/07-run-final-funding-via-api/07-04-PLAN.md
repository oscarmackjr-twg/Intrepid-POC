```python
from sqlalchemy import Column, Integer, String, DateTime, ForeignKey, JSON

class AuditLog(Base):

    id         = Column(Integer, primary_key=True, index=True)
    event_type = Column(String(100), nullable=False, index=True)
    user_id    = Column(Integer, ForeignKey("users.id", ondelete="SET NULL"), nullable=True, index=True)
```

```python
from db.models import AuditLog
from db.database import SessionLocal  # import for background-task use

def log_user_action(action: str, user=None, db=None, source_ip=None, resource=None, outcome="success", details=None):
    # 1. Existing logger write (preserve)
    logger.info("audit: action=%s user=%s outcome=%s source_ip=%s resource=%s",
                action, getattr(user, 'id', None), outcome, source_ip, resource)
    # 2. DB write when session provided
    if db is not None:
        try:
            entry = AuditLog(
                event_type=action,
                user_id=getattr(user, 'id', None),
                source_ip=source_ip,
                resource=resource,
                outcome=outcome,
                detail_json=details or {},
            )
            db.add(entry)
            db.commit()
        except Exception as e:
            logger.error("Failed to write audit log to DB: %s", e)
            # Do NOT re-raise — audit failure must not break the request
```
